def calcular_clases(datos):
    clases = sorted(set(datos))  # Obtener clases únicas ordenadas
    return clases

def asignar_indices(datos):
    clases_unicas = calcular_clases(datos)
    clase_a_indice = {clase: idx for idx, clase in enumerate(clases_unicas)}
    indices_clases = [clase_a_indice[elem] for elem in datos]
    return indices_clases, clase_a_indice

def calcular_limites(clases):
    limite_inferior = 3
    limite_superior = limite_inferior + 1
    limites_inferiores = []
    limites_superiores = []
    while limite_superior <= 9:
        limites_inferiores.append(limite_inferior)
        limites_superiores.append(limite_superior)
        limite_inferior += 1
        limite_superior += 1
    return limites_inferiores, limites_superiores

def calcular_marcas(limites_inferiores, limites_superiores):
    marcas_clase = [(li + ls) / 2 for li, ls in zip(limites_inferiores, limites_superiores)]
    return marcas_clase

def calcular_frecuencias(datos, clases):
    frecuencias_clase = [0] * len(clases)
    for dato in datos:
        indice_clase = clases.index(dato)
        frecuencias_clase[indice_clase] += 1
    return frecuencias_clase

def calcular_frecuencias_relativas(frecuencias_clase, total_datos):
    frecuencias_relativas = [f / total_datos for f in frecuencias_clase]
    return frecuencias_relativas

def calcular_frecuencias_acumuladas(frecuencias_relativas):
    frecuencias_acumuladas = []
    acumulado = 0
    for frec in frecuencias_relativas:
        acumulado += frec
        frecuencias_acumuladas.append(acumulado)
    return frecuencias_acumuladas


# Calcular clases y asignar índices
indices_clases, clase_a_indice = asignar_indices(datos)
clases = list(clase_a_indice.keys())

# Calcular límites inferiores y superiores
limites_inferiores, limites_superiores = calcular_limites(clases)

# Calcular marcas de clase
marcas_clase = calcular_marcas(limites_inferiores, limites_superiores)

# Calcular frecuencias
frecuencias_clase = calcular_frecuencias(datos, clases)

# Calcular frecuencias relativas
total_datos = len(datos)
frecuencias_relativas = calcular_frecuencias_relativas(frecuencias_clase, total_datos)

# Calcular frecuencias acumuladas con frecuencias relativas
frecuencias_acumuladas = calcular_frecuencias_acumuladas(frecuencias_relativas)

# Formatear y mostrar los resultados en una tabla

for i in range(len(clases)):
